#comment -------------------------------------------------------------------
#comment TOPM - skip # lines at top
#comment LEFTM - left margin (# chars)
#comment RIGHTM - right margin (#chars)
#comment COLS - numer of columns (total, including margins)
#comment PAGELINES - number of lines on page

#define TOPM			0
#define LEFTM			0
#define RIGHTM			0
#define COLS			120
#define PAGELINES		63
#comment -------------------------------------------------------------------


<%
class cgenerator
#comment '-------------------------------
#comment ' document header
#comment '-------------------------------
	private sub generate_header()
		page.set_cpi 15
		page.rwriteat COLS-RIGHTM,util_.Firma_Miasto + " " +  fulldate(root_.dDataWyst)
		nl
		nl
		page.double_width true
		page.double_height true
		page.writeat LEFTM, "FAKTURA VAT " + root_.text
		page.double_height false
		page.double_width false
		nl
		page.set_cpi 15
		dim oHdrTW : set oHdrTW = new ctabwriter : oHdrTW.init Me
		oHdrTW.make_col 1,12 'sprz=label
		oHdrTW.make_col 2,42 'sprz=data
		oHdrTW.make_col 3,13 'nab=label
		oHdrTW.make_col 4,47 'nab=data

		oHdrTW.write	1,"Sprzedawca:"
		oHdrTW.box		2, util_.Firma_Nazwa
		oHdrTW.write	3,"Nabywca:"
		oHdrTW.box		4, util_.Klient_Nazwa(root_.pKlient)
		nl
		oHdrTW.write	1,"Adres:"
		oHdrTW.box		2, util_.Firma_Ulica
		oHdrTW.write	3,"Adres:"
		oHdrTW.box		4, util_.Klient_Ulica(root_.pKlient)
		nl
		oHdrTW.box		2, util_.Firma_Kod_Miasto
		oHdrTW.box		4, root_.pKlient.strKod + " " + root_.pKlient.pMiejscowosc.strNazwa 
		nl

		oHdrTW.write	1,"NIP:"
		oHdrTW.write	2, util_.Firma_NIP
		oHdrTW.write	3,"NIP:"
		oHdrTW.write	4, util_.Klient_NIP(root_.pKlient)
		newlines 2
		
		oHdrTW.write	1,"Data sprzed.:"
		oHdrTW.write	2, fulldate(root_.dDataSprzed)
		oHdrTW.write	3,"Sposób płatn.:"
		oHdrTW.write	4, util_.Dokument_Sposob_Platnosci(root_)
		nl

		oHdrTW.write	3,"Termin płatn.:"
		oHdrTW.write	4, fulldate(root_.dDataPlat)
		nl

		oHdrTW.write	1,"Zamówienie:"
		oHdrTW.write	2, root_.pZamowienie.text
		oHdrTW.write	3,"Wasz nr zam.:"
		oHdrTW.write	4, root_.pZamowienie.zam_strNrDokObcy
		newlines 2
	end sub

#comment '-------------------------------
#comment ' document footer
#comment '-------------------------------
	private sub generate_footer()
		page.set_cpi 10
		nl
		page.double_width true
		if is_koncowa then
			page.writeat LEFTM, "Do zaplaty: " + money(root_.curSumaBrutto) + " " + root_.pWaluta.strSkrot
		else
			page.writeat LEFTM, "Kwota zaliczki brutto: " + money(root_.curSumaBrutto) + " " + root_.pWaluta.strSkrot
		end if
		nl
		page.double_width false
		page.set_cpi 15
		page.writeat LEFTM, "Słownie: " +  doc_.afc.display.slownie(root_.curSumaBrutto)
		newlines 2

		if is_koncowa then
			generate_poprzednie_fzal
		end if
		newlines 2

		if root_.strUwagi <> "" then
			page.writeat LEFTM,"Adnotacje"
			nl
			hline1 : nl
			page.set_cpi 12
			page.condensed true
			page.box LEFTM,COLS-LEFTM-RIGHTM, root_.strUwagi
			page.condensed false
			page.set_cpi 15
			nl
			hline1 : nl
		end if

		newlines 5

		page.writeat LEFTM + 20,root_.pUzytRej.text
		page.writeat LEFTM + 80,util_.Osoba_Odbierajaca(root_)

		nl
		page.set_cpi 12
		page.condensed true
		'kropki
		hline_ex LEFTM,45,"."
		hline_ex LEFTM+86,45,"."
		nl
		if is_koncowa then
			page.box LEFTM,45,"Imię i nazwisko osoby upoważnionej do wystawienia faktury zaliczkowej końcowej"
			page.box LEFTM+86,45,"Imię i nazwisko osoby uprawnionej do odbioru faktury zaliczkowej końcowej"
		else
			page.box LEFTM,45,"Imię i nazwisko osoby upoważnionej do wystawienia faktury zaliczkowej"
			page.box LEFTM+86,45,"Imię i nazwisko osoby uprawnionej do odbioru faktury zaliczkowej"
		end if
		newlines 3
		page.writeat LEFTM,"Dokument wyemitowany przez EuroBusiness 5.0"
		page.condensed false
		page.set_cpi 15	
	end sub
#comment '-------------------------------
#comment ' positions (table)
#comment '-------------------------------
	private sub generate_table_header()
		page.set_cpi 15
		'----------------------
		'-- write header
		hline1 : nl
		m_oPozTW.write  1,"LP"
		m_oPozTW.write  2,"Nazwa"
		m_oPozTW.write  3,"Indeks"
		m_oPozTW.rwrite 4,"Ilość"
		m_oPozTW.rwrite 5,"Cena"
		m_oPozTW.rwrite 6,"W. netto"
		m_oPozTW.rwrite 7,"VAT"
		m_oPozTW.rwrite 8,"W. VAT"
		m_oPozTW.rwrite 9,"W. brutto"
		nl : hline2 : nl
	end sub
	private sub generate_table()
		m_bInTable = true
		generate_table_header
		'----------------------
		'-- positions

		'flaga dla kolejnych stawek VAT - czy drukowac je na podsumowaniu
		dim oVATArray:set oVATArray = trans_.afc.new_navoarray()
		dim iter : iter = 1
		dim oDocument
		if is_koncowa then
			set oDocument = root_
		else
			set oDocument = root_.pZamowienie
		end if
		dim oPozycja
		for each oPozycja in oDocument.listPozycje
			generate_table_line iter,oPozycja
			nl
			iter = iter + 1

			if not oVATArray.exists(clng(oPozycja.eVat)) then
				oVATArray.add(clng(oPozycja.eVat))
			end if
		next
		m_bInTable = false
		hline2 : nl
		nl
		if is_koncowa then
			m_oPozTW.rwrite 5,"RAZEM:"
		else
			'----------------------
			hline_ex m_oPozTW.get_pos(6),COLS - m_oPozTW.get_pos(6),"="
			nl
			'----------------------
			'-- summary
			m_oPozTW.rwrite 5,"WPŁACANA ZALICZKA:"
		end if
		m_oPozTW.rwrite 6,money(root_.curSumaNetto)
		m_oPozTW.rwrite 7,"x"
		m_oPozTW.rwrite 8,money(root_.curSumaVat)
		m_oPozTW.rwrite 9,money(root_.curSumaBrutto)
		nl
		'----------------------
		for iter=0 to oVATArray.UpperIndex()
			m_oPozTW.rwrite 6,util_.netto_dla_stawki(oVATArray.value(iter),root_)
			m_oPozTW.rwrite 7,util_.vat_to_name(oVATArray.value(iter))
			m_oPozTW.rwrite 8,util_.vat_dla_stawki(oVATArray.value(iter),root_)
			m_oPozTW.rwrite 9,util_.brutto_dla_stawki(oVATArray.value(iter),root_)
			nl
		next
		'----------------------
		hline_ex m_oPozTW.get_pos(6),COLS - m_oPozTW.get_pos(6),"="
		nl
	end sub

	public sub generate_table_line(byval iLP,byref oPozycja)
		m_oPozTW.write	1,cstr(iLP)
		m_oPozTW.box		2,oPozycja.pProdukt.strNazwa
		m_oPozTW.box		3,oPozycja.pProdukt.strIndeks
		m_oPozTW.rwrite	4,qty_unit(oPozycja.curIloscOpak,oPozycja.pOpak.pJednOpak)
		m_oPozTW.rwrite	5,qty(oPozycja.curCenaNetto,"",trans_.var(GVAR_PRICE_PREC))
		m_oPozTW.rwrite	6,money(oPozycja.curWartNetto)
		m_oPozTW.rwrite	7,util_.vat_to_name(oPozycja.eVat)
		m_oPozTW.rwrite	8,money(oPozycja.curVAT)
		m_oPozTW.rwrite	9,money(oPozycja.curWartBrutto)
	end sub
		
	public sub generate_body()
		m_iPageNr = 1
		page.footer_line = PAGELINES
		newlines TOPM
		'--- define columns ---
		set m_oPozTW = new ctabwriter : m_oPozTW.init Me
		m_oPozTW.make_col 1,3	'lp
		m_oPozTW.make_col 2,35-LEFTM	'nazwa
		m_oPozTW.make_col 3,12	'indeks
		m_oPozTW.make_col 4,10	'ilosc
		m_oPozTW.make_col 5,10	'cena
		m_oPozTW.make_col 6,12	'w.netto
		m_oPozTW.make_col 7,4		'st.vat
		m_oPozTW.make_col 8,12	'w.vat
		m_oPozTW.make_col 9,12	'w.brutto

		generate_header
		generate_table
		generate_footer
	end sub

	public sub generate_poprzednie_fzal()
		dim oFZalRS : set oFZalRS = trans_.createrawcursor(TYPEID_N5SPRZEDAZ, _
			"pZamowienie=" + cstr(root_.pZamowienie.idobj) + " and idobj<> " + cstr(root_.idobj),"dDataWyst,idobj","")
		dim oFraZal
		dim bFirst : bFirst = true
		for each oFraZal in oFZalRS
			if bFirst then
				page.writeat LEFTM,"Poprzednie faktury zaliczkowe"
				nl
				hline_ex LEFTM,45,"-"
				nl
				page.writeat LEFTM, "Numer faktury"
				page.writeat LEFTM + 20, "Data wyst."
				page.rwriteat LEFTM + 45, "W. brutto"
				nl
				hline_ex LEFTM,45,"-"
				nl
				bFirst = false
			end if
			page.writeat LEFTM, oFraZal.strNrDok
			page.writeat LEFTM + 20, fulldate(oFraZal.dDataWyst)
			page.rwriteat LEFTM + 45, money(oFraZal.curSumaBrutto)
			nl
		next
		if not bFirst then
			hline_ex LEFTM,45,"-"
			nl
		end if
	end sub

	public sub on_page_footer()
		hline1 : nl
		page.condensed true
		page.write "Faktura VAT " + root_.text + ",Strona nr " + cstr(m_iPageNr)
		page.condensed false

		m_oESC.new_page().footer_line = PAGELINES
		
		if m_bInTable then
			generate_table_header
		end if
		
		m_iPageNr = m_iPageNr + 1
	end sub


	private property get is_koncowa
		is_koncowa = (root_.listPozycje.count > 0)
	end property
#comment '-------------------------------------------------------------------
#comment '-- common part (version 1)
#comment '-------------------------------------------------------------------

	public sub init(byref oESC)
		set m_oESC = oESC
	end sub 

	public property get page
		set page = m_oESC.page
	end property

#comment '-------------------------------------------------------------------
#comment 'print iCount empty lines
#comment '-------------------------------------------------------------------
	private sub newlines(iCount)
		if iCount > 0 then
			page.nextline
			iCount = iCount - 1
			do while iCount > 0
				page.newline
				iCount = iCount - 1
			loop
		end if
	end sub

	private sub nl
		newlines 1
	end sub

#comment '-------------------------------------------------------------------
#comment 'print horizontal separator
#comment '-------------------------------------------------------------------
	private sub hline_ex(iFrom,iCount,cChar)
		page.writeat iFrom,string(iCount,cChar)
	end sub

	private sub hline1
		hline_ex LEFTM,COLS-LEFTM-RIGHTM,"-"
	end sub
	
	private sub hline2
		hline_ex LEFTM,COLS-LEFTM-RIGHTM,"="
	end sub

#comment '-------------------------------------------------------------------
#comment ' formatting helpers
#comment '-------------------------------------------------------------------
	private function fulldate(dData)
		fulldate = doc_.afc.display.fulldate(dData)
	end function
	private function money(curValue)
		money = doc_.afc.display.money(curValue,"")
	end function
	private function qty_unit(curValue,pUnit)
		qty_unit = qty(curValue,pUnit.strSkrot,pUnit.nDokladnosc-1)
	end function
	private function qty(curValue,strUnit,iPrec)
		qty = doc_.afc.display.quantity(curValue,strUnit,iPrec)
	end function

#comment '-------------------------------------------------------------------
#comment ' members
#comment '-------------------------------------------------------------------
	private m_oESC
	private m_oPozTW
	private m_bInTable
	private m_iPageNr
end class

class ctabwriter
	public sub init(byref oGenerator)
		set m_oGenerator = oGenerator
		m_oTabs(0) = LEFTM
	end sub

	public sub make_col(iCol,iWidth)
		m_oTabs(iCol) = iWidth
	end sub
	public sub write(iCol,strText)
		m_oGenerator.page.writeat get_pos(iCol),strText
	end sub
	public sub rwrite(iCol,strText)
		m_oGenerator.page.rwriteat get_pos(iCol+1)-1,strText
	end sub
	public sub box(iCol,strText)
		m_oGenerator.page.box get_pos(iCol),m_oTabs(iCol),strText
	end sub

	public function get_pos(iCol)
		dim iPos : iPos = 0
		dim iter
		for iter = 0 to iCol - 1
			iPos = iPos + m_oTabs(iter) + 1
		next
		get_pos = iPos
	end function

	private m_oTabs(50)
	private m_oGenerator
end class

dim oGenerator : set oGenerator = new cgenerator
oGenerator.init esc_
esc_.set_callback oGenerator
oGenerator.generate_body
