function query()
query = "select 'c_mag' = max(t_mag.strNazwa),'c_field_second' = C_FIELD_SECOND_EXPR,'c_field_first'= C_FIELD_FIRST_EXPR, " +_
		"'c_ilosc_opak'=isnull((select sum(gmo_curilosc) from n5gmop t_0 where t_0.gmo_pmag = t_mag.idobj and t_0.gmo_pprod=t_prod.idobj and NUMOPBORDER),0), " +_
		"'c_wartosc'=isnull((select sum(gmo_curwartosc) from n5gmop t_0 where t_0.gmo_pmag = t_mag.idobj and t_0.gmo_pprod=t_prod.idobj and NUMOPBORDER),0), " +_
		"'c_prod_idobj'=t_prod.idobj,'c_jedn_opak'=max(t_jedn.strSkrot),'c_prec_opak' = max(t_jedn.nDokladnosc) " +_
		" from n5magazyn t_mag cross join n5produkt t_prod " +_
		" inner join n5jednostka as t_jedn on t_prod.pJednostkaEwid = t_jedn.IdObj " +_
		" inner join n5GrupaProduktow as t_grupy on t_prod.pGrupaProd = t_grupy.idobj " + _
		" where %1 and %3 and %4 and %5 " +_
		" group by t_mag.idobj,t_prod.idobj " +_
		" order by c_field_first,c_field_second,c_mag"
end function

function write_actual_param(byref pActualParam)
dim iter
dim bAnyWritten : bAnyWritten = false
if not pActualParam is nothing then
for iter = 0 to pActualParam.upperIndex
if pActualParam.value(iter).get_desc() <> "" then 
bAnyWritten = true
	so.write "<t size='5' /><text fontsize='4'><![CD"+"ATA[" + pActualParam.value(iter).get_desc() + "]"+"]></text><br/>"
end if
next
end if
if not bAnyWritten then 
	so.write "<t fontsize='4'>brak</t><br/>"
end if
	so.write "<br/>"
end function
%>
<%
dim g_idLastProd : g_idLastProd = clng(0)
function body_row(byref pSection,byref pRec)
	
	dim idProd : idProd = clng(pRec("c_prod_idobj"))
	if idProd <> g_idLastProd then
		finish_body_row
		g_idLastProd = idProd
		begin_body_row pRec
	end if
%>
<td align='right'>
	RXE_QUANTITY_VVV_NON0(ccur(pRec("c_ilosc_opak")),pRec("c_jedn_opak"),pRec("c_prec_opak"))
</td>
<%
end function

	</td>
	<td>

	</td>
<%
end sub
